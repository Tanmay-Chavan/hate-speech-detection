hadoop fs -rmr \*
hadoop namenode -format
start-all.sh
java -cp target/hate-speech-detection-0.0.1-SNAPSHOT-jar-with-dependencies.jar com.tanmay.dads.hate_speech_detection.TweetReader data/hate_speech_train.tsv tweets-seq

mahout testnb -i test-vectors -m model -l labelindex -ow -o tweets-testing -c
